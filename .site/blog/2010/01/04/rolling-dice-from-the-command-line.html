puts eval(ARGV.join.gsub(/(\d+)d(\d+)/) {|s|
  rolls = $1.to_i.times.map { rand($2.to_i) + 1 }
  sum = rolls.inject {|n,i| n + i }
  puts "#{s}\t#{sum}\t#{rolls.join(', ')}"
  sum
})
